 
   PRINT 'ACTUALIZANDO FTRCXC'
 DECLARE FTRCXCUR CURSOR FOR 
   SELECT DISTINCT N_FACTURA FROM FCXCD WHERE CNSCXC=@CNSCXC
   OPEN FTRCXCUR
   FETCH NEXT FROM FTRCXCUR
   INTO @N_FACTURA1
   WHILE @@FETCH_STATUS = 0  
   BEGIN 
      UPDATE FTRCXC SET CUENTACXC=A.CUENTACXC_RAD
      FROM FTR A INNER JOIN FTRCXC B ON A.N_FACTURA=B.N_FACTURA
      WHERE A.N_FACTURA=@N_FACTURA1
      FETCH NEXT FROM FTRCXCUR  
      INTO @N_FACTURA1  
   END  
   CLOSE FTRCXCUR  
   DEALLOCATE FTRCXCUR
